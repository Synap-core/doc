export function useChatMessages(
  chatId: string,

) {
  return trpc.chat.getMessages.useInfiniteQuery(
    { chatId, limit },

**2. `useSendMessage.ts`**

```typescript
/**
 * Send chat messages with optimistic updates
 */
export function useSendMessage(chatId: string) {
  return trpc.chat.sendMessage.useMutation({
    onMutate: async (newMessage) => {
      // Optimistic update
      utils.chat.getMessages.setInfiniteData({ chatId }, (old) => ({
        ...old,
        pages: [
          {
            messages: [
              ...old.pages[0].messages,
              { id: "temp", role: "user", content: newMessage.content },
            ],
          },
        ],
      }));
    },
    onSuccess: () => {
      utils.chat.getMessages.invalidate({ chatId });
    },
  });
}
```

**Status**: ✅ Exists  
**What it does**: Sends messages with optimistic UI  
**Gap**: No support for streaming, AI steps, tool execution

## 📊 Data Flow Architecture

### Current Flow (Simple Chat)

```
User Input
    ↓
useSendMessage() → tRPC → Backend
    ↓
Optimistic Update (local)
    ↓
Real Response → useChatMessages() → UI Update
```

### Needed Flow (AI Chat with Steps)

```
User Input
    ↓
useSendMessage() → tRPC → Backend → Intelligence Service
    ↓                                      ↓
Optimistic Update                    AI Processing
    ↓                                      ↓
useStreamingMessage()  ←─────  Real-time Events (Socket.IO)
    ↓                                      ↓
Streaming UI                         AI Steps Emitted:
    ├─ Thinking: "Analyzing..."           - thinking
    ├─ Tool Call: "Searching..."          - tool_call
    ├─ Tool Result: "Found 5 docs"        - tool_result
    ├─ Verification: "Create task?"       - verification
    └─ Final Response                     - complete
         ↓
    useChatMessages() → Full Message with Steps
```
