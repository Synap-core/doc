# Signal Feed AI Classification & Memory Integration

## Overview

The Signal Feed uses the pod's Intelligence Service for AI classification and integrates with the existing memory system for personalization. This document covers how AI classification works, how memory is used, and how signals integrate with proactive intelligence.

### 2. Relevance Scoring

```typescript
function computeRelevanceScore(
  item: ClassifiedItem,
  memory: SignalMemory,
  context: ClassificationContext
): number {
  let score = 0;
  let totalWeight = 0;

  // 1. Explicit subscriptions (40% weight)
  const subscriptionWeight = 0.4;
  for (const sub of memory.subscriptions) {
    const match = item.topics.find(t => t.name === sub.topic);
    if (match) {
      score += match.confidence * sub.confidence * subscriptionWeight;
      totalWeight += subscriptionWeight;
    }
  }

  // 2. Implicit classifications (30% weight)
  const classificationWeight = 0.3;
  for (const cls of memory.classifications) {
    const match = item.topics.find(t => t.name === cls.topic);
    if (match) {
      // Apply time decay
      const daysSinceUpdate = daysBetween(cls.lastUpdated, new Date());
      const decayedConfidence = cls.confidence * Math.pow(0.95, daysSinceUpdate);

      score += match.confidence * decayedConfidence * classificationWeight;
      totalWeight += classificationWeight;
    }
  }

  // 3. Current focus match (20% weight)
  const focusWeight = 0.2;
  if (context.currentFocus) {
    const focusMatches = countFocusMatches(item, context.currentFocus);
    if (focusMatches > 0) {
      score += (focusMatches / 3) * focusWeight; // Max 3 focus areas
      totalWeight += focusWeight;
    }
  }

  // 4. Recency boost (10% weight)
  const recencyWeight = 0.1;
  const hoursOld = hoursBetween(new Date(item.publishedAt), new Date());
  const recencyBoost = Math.max(0.5, 1 - (hoursOld / 168)); // 1 week decay
  score += recencyBoost * recencyWeight;
  totalWeight += recencyWeight;

  // Normalize to 0-1
  return totalWeight > 0 ? Math.min(1, score / (totalWeight * 0.8)) : 0.1;
}

function countFocusMatches(item: ClassifiedItem, focus: CurrentFocus): number {
  let matches = 0;

  // Check active tasks (extract keywords)
  if (focus.activeTasks) {
    const taskText = focus.activeTasks.join(' ').toLowerCase();
    for (const topic of item.topics) {
      if (taskText.includes(topic.name)) matches++;
    }
  }

  // Check recent searches
  if (focus.recentSearches) {
    const searchText = focus.recentSearches.join(' ').toLowerCase();
    for (const topic of item.topics) {
      if (searchText.includes(topic.name)) matches++;
    }
  }

  // Check discussed people
  if (focus.discussedPeople && item.entities) {
    for (const entity of item.entities) {
      if (entity.type === 'person' && focus.discussedPeople.includes(entity.name)) {
        matches++;
      }
    }
  }

  return matches;
}
```

```typescript
async function getSignalMemory(
  userId: string,
  workspaceId: string
): Promise<SignalMemory> {
  const [subscriptions, classifications] = await Promise.all([
    db.query(`
      SELECT * FROM signal_subscriptions
      WHERE user_id = $1 AND workspace_id = $2 AND is_active = true
    `, [userId, workspaceId]),

    db.query(`
      SELECT * FROM signal_classifications
      WHERE user_id = $1 AND workspace_id = $2
      AND confidence > 0.05
      ORDER BY confidence DESC
      LIMIT 50
    `, [userId, workspaceId]),
  ]);

  return {
    subscriptions: subscriptions.rows,
    classifications: classifications.rows.map(c => ({
      ...c,
      confidence: c.confidence * Math.pow(c.decay_rate,
        daysBetween(new Date(c.last_decay_at), new Date())),
    })),
  };
}
```

## Feed Personalization Algorithm

### Ranking Formula

```typescript
function rankFeedItems(
  items: ClassifiedSignalItem[],
  memory: SignalMemory,
  context: FeedContext
): RankedItem[] {
  const scored = items.map(item => {
    let score = 0;

    // Base relevance from classification (0-1)
    score += item.relevanceScore * 0.4;

    // Subscription match (0-1)
    const subscriptionScore = calculateSubscriptionScore(item, memory);
    score += subscriptionScore * 0.25;

    // Classification match (0-1)
    const classificationScore = calculateClassificationScore(item, memory);
    score += classificationScore * 0.15;

    // Recency (0-1, exponential decay)
    const hoursOld = hoursBetween(new Date(item.originalItem.publishedAt), new Date());
    const recencyScore = Math.exp(-hoursOld / 72); // 72h half-life
    score += recencyScore * 0.1;

    // Engagement prediction (0-1)
    const engagementScore = predictEngagement(item, context);
    score += engagementScore * 0.1;

    return {
      ...item,
      finalScore: score,
      scoreBreakdown: {
        relevance: item.relevanceScore * 0.4,
        subscription: subscriptionScore * 0.25,
        classification: classificationScore * 0.15,
        recency: recencyScore * 0.1,
        engagement: engagementScore * 0.1,
      },
    };
  });

  // Apply diversification
  const diversified = diversifyResults(scored, {
    targetTopics: memory.subscriptions.map(s => s.topic).slice(0, 3),
    maxFromSameSource: 3,
    maxFromSameTopic: 5,
  });

  return diversified.sort((a, b) => b.finalScore - a.finalScore);
}

function calculateSubscriptionScore(item: ClassifiedSignalItem, memory: SignalMemory): number {
  if (memory.subscriptions.length === 0) return 0.5;

  let score = 0;
  let matches = 0;

  for (const sub of memory.subscriptions) {
    const topicMatch = item.topics.find(t => t.name === sub.topic);
    if (topicMatch) {
      score += topicMatch.confidence * sub.confidence;
      matches++;
    }

    // Platform match
    if (item.originalItem.sourcePlatform === sub.sourcePlatform) {
      score += 0.2 * sub.confidence;
    }
  }

  return matches > 0 ? score / matches : 0;
}

function calculateClassificationScore(item: ClassifiedSignalItem, memory: SignalMemory): number {
  if (memory.classifications.length === 0) return 0.5;

  let score = 0;
  let matches = 0;

  for (const cls of memory.classifications) {
    const topicMatch = item.topics.find(t => t.name === cls.topic);
    if (topicMatch) {
      // Apply time decay
      const daysSinceUpdate = daysBetween(new Date(cls.lastSeenAt), new Date());
      const decayedConfidence = cls.confidence * Math.pow(0.95, daysSinceUpdate);

      score += topicMatch.confidence * decayedConfidence;
      matches++;
    }
  }

  return matches > 0 ? score / matches : 0;
}

function predictEngagement(item: ClassifiedSignalItem, context: FeedContext): number {
  let score = 0.5; // Base score

  // Has summary (easier to consume)
  if (item.aiSummary) score += 0.1;

  // Has image (more engaging)
  if (item.originalItem.enclosure?.url) score += 0.1;

  // From trusted platforms
  const trustedPlatforms = ['hackernews', 'github', 'producthunt'];
  if (trustedPlatforms.includes(item.originalItem.sourcePlatform)) {
    score += 0.1;
  }

  // Quality signals (longer content typically higher quality)
  const contentLength = (item.originalItem.content || '').length;
  if (contentLength > 500) score += 0.1;
  else if (contentLength < 100) score -= 0.1;

  return Math.max(0, Math.min(1, score));
}

function diversifyResults(
  items: RankedItem[],
  options: {
    targetTopics: string[];
    maxFromSameSource: number;
    maxFromSameTopic: number;
  }
): RankedItem[] {
  const result: RankedItem[] = [];
  const sourceCount: Record<string, number> = {};
  const topicCount: Record<string, number> = {};

  for (const item of items) {
    const source = item.originalItem.sourcePlatform;
    const primaryTopic = item.topics[0]?.name || 'other';

    // Check limits
    const sourceOk = (sourceCount[source] || 0) < options.maxFromSameSource;
    const topicOk = (topicCount[primaryTopic] || 0) < options.maxFromSameTopic;

    if (sourceOk && topicOk) {
      result.push(item);
      sourceCount[source] = (sourceCount[source] || 0) + 1;
      topicCount[primaryTopic] = (topicCount[primaryTopic] || 0) + 1;
    } else if (result.length < 10) {
      // Always include top 10 regardless of diversification
      result.push(item);
    }
  }

  return result;
}
```

```typescript
async function getClassificationContext(
  userId: string,
  workspaceId: string
): Promise<ClassificationContext> {
  const [
    recentTasks,
    recentSearches,
    discussedPeople,
    recentViews,
    sessionMemory,
  ] = await Promise.all([
    // Recent tasks (last 7 days, active or recently updated)
    db.query(`
      SELECT id, name, properties
      FROM entities
      WHERE workspace_id = $1
        AND profile_slug = 'task'
        AND (properties->>'status' IN ('in-progress', 'todo')
             OR updated_at > NOW() - INTERVAL '7 days')
      ORDER BY updated_at DESC
      LIMIT 5
    `, [workspaceId]),

    // Recent searches (from audit logs)
    db.query(`
      SELECT query
      FROM audit_logs
      WHERE user_id = $1
        AND action = 'search'
        AND created_at > NOW() - INTERVAL '7 days'
      ORDER BY created_at DESC
      LIMIT 10
    `, [userId]),

    // People mentioned in recent messages
    db.query(`
      SELECT DISTINCT e.id, e.name, e.profile_slug as type
      FROM messages m
      JOIN entities e ON e.name = m.content  -- Simplified
      WHERE m.workspace_id = $1
        AND m.created_at > NOW() - INTERVAL '3 days'
        AND e.profile_slug IN ('person', 'contact')
      LIMIT 10
    `, [workspaceId]),

    // Recent entity views
    db.query(`
      SELECT DISTINCT e.id, e.name, e.profile_slug as type
      FROM entity_views ev
      JOIN entities e ON e.id = ev.entity_id
      WHERE ev.workspace_id = $1
        AND ev.viewed_at > NOW() - INTERVAL '7 days'
      ORDER BY ev.viewed_at DESC
      LIMIT 20
    `, [workspaceId]),

    // Session memory from IS
    getSessionMemory(userId),
  ]);

  return {
    currentFocus: {
      activeTasks: recentTasks.rows.map(t => t.name),
      recentSearches: recentSearches.rows.map(r => r.query),
      discussedPeople: discussedPeople.rows.map(p => ({ id: p.id, name: p.name })),
      viewedEntities: recentViews.rows.map(v => ({ id: v.id, name: v.name, type: v.type })),
    },
    sessionMemory,
  };
}
```

## Performance Optimization

### Batch Classification

```typescript
async function classifySignalsBatch(
  items: RssHubItem[],
  context: ClassificationContext,
  options: {
    batchSize?: number;      // Default: 10
    maxConcurrency?: number; // Default: 3
  }
): Promise<ClassifiedSignalItem[]> {
  const batchSize = options.batchSize || 10;
  const maxConcurrency = options.maxConcurrency || 3;

  const results: ClassifiedSignalItem[] = [];

  // Process in batches
  for (let i = 0; i < items.length; i += batchSize) {
    const batch = items.slice(i, i + batchSize);

    // Process batch with concurrency limit
    const batchResults = await Promise.all(
      batch.slice(0, maxConcurrency).map(item =>
        classifySingleSignal(item, context)
      )
    );

    results.push(...batchResults);

    // Small delay between batches to avoid rate limits
    if (i + batchSize < items.length) {
      await sleep(100);
    }
  }

  return results;
}
```

### Caching Strategy

```typescript
// Cache classification results for similar items
const classificationCache = new LRUCache<string, ClassifiedSignalItem>({
  max: 1000,
  ttl: 3600000, // 1 hour
});

function getCacheKey(item: RssHubItem): string {
  // Create hash of title + source + date
  const data = `${item.title}|${item.sourcePlatform}|${item.publishedAt}`;
  return createHash('sha256').update(data).digest('hex').substring(0, 16);
}

async function classifyWithCache(
  item: RssHubItem,
  context: ClassificationContext
): Promise<ClassifiedSignalItem> {
  const cacheKey = getCacheKey(item);

  if (classificationCache.has(cacheKey)) {
    return classificationCache.get(cacheKey)!;
  }

  const result = await classifySingleSignal(item, context);
  classificationCache.set(cacheKey, result);

  return result;
}
```
