# Synap Browser Evolution - Implementation Plan

## Executive Summary

This plan transforms Synap Browser from a basic workspace browser into a **full workspace OS** with:
- Multi-pod auth with profile-pod linking
- Dashboard views stored in pod (not local)
- Extensible widget system with graceful error handling
- Expanded entity profiles for CRM, captures, etc.
- AI-powered app generation foundation

```typescript
  extendedRegistry[key] = component;
}

  return CORE_WIDGETS[key] ?? extendedRegistry[key] ?? null;

  return key in CORE_WIDGETS || key in extendedRegistry;
```

## Phase 3: Auth Architecture Upgrade

### 3.1 Multi-Session Auth Manager

**Goal**: Support multiple pod sessions + ecosystem identity

**File**: `/browser/electron/main/auth/auth-manager.ts`

```typescript
import { session } from 'electron';
import log from 'electron-log/main';

interface EcosystemIdentity {
  userId: string;
  email: string;
  subscription?: {
    tier: string;
    status: 'active' | 'past_due' | 'canceled';
  };
}

interface PodSession {
  podUrl: string;
  userId: string;        // User ID in THIS pod
  workspaces: { id: string; name: string; role: string }[];
  sessionCookie: string;
  expiresAt?: Date;
}

export class AuthManager {
  private defaultSession = session.defaultSession;

  // Ecosystem (Control Plane) identity
  private ecosystemIdentity: EcosystemIdentity | null = null;

  // Active pod sessions (can have multiple)
  private podSessions: Map<string, PodSession> = new Map();

  // Currently active pod
  private activePodUrl: string | null = null;

  constructor() {
    log.info('[AuthManager] Initialized');
  }

  // ========== Ecosystem Identity ==========

  async getEcosystemIdentity(): Promise<EcosystemIdentity | null> {
    if (this.ecosystemIdentity) return this.ecosystemIdentity;

    // Try to fetch from control plane
    const controlPlaneUrl = process.env.CONTROL_PLANE_URL || 'https://app.synap.live';

    try {
      const cookies = await this.defaultSession.cookies.get({
        url: controlPlaneUrl,
        name: 'ory_kratos_session'
      });

      if (cookies.length === 0) return null;

      // Fetch whoami from control plane
      const response = await fetch(`${controlPlaneUrl}/.ory/kratos/public/sessions/whoami`, {
        headers: { Cookie: `ory_kratos_session=${cookies[0].value}` }
      });

      if (!response.ok) return null;

      const session = await response.json();

      this.ecosystemIdentity = {
        userId: session.identity.id,
        email: session.identity.traits.email,
      };

      return this.ecosystemIdentity;
    } catch (err) {
      log.error('[AuthManager] Failed to get ecosystem identity:', err);
      return null;
    }
  }

  async signInToEcosystem(): Promise<{ success: boolean; identity?: EcosystemIdentity }> {
    // Opens BrowserWindow to control plane login
    return new Promise((resolve) => {
      const controlPlaneUrl = process.env.CONTROL_PLANE_URL || 'https://app.synap.live';

      const authWindow = new BrowserWindow({
        width: 480,
        height: 640,
        webPreferences: {
          session: this.defaultSession,
        }
      });

      authWindow.loadURL(`${controlPlaneUrl}/auth`);

      // Watch for session cookie
      const cookieWatcher = async (_event, cookie, _cause, removed) => {
        if (cookie.name === 'ory_kratos_session' && !removed) {
          this.defaultSession.cookies.off('changed', cookieWatcher);
          authWindow.close();

          const identity = await this.getEcosystemIdentity();
          resolve({ success: true, identity: identity || undefined });
        }
      };

      this.defaultSession.cookies.on('changed', cookieWatcher);

      authWindow.on('closed', () => {
        this.defaultSession.cookies.off('changed', cookieWatcher);
        resolve({ success: false });
      });
    });
  }

  // ========== Pod Sessions ==========

  async connectToPod(podUrl: string): Promise<{ success: boolean; session?: PodSession }> {
    // Check if we already have a valid session
    const existing = this.podSessions.get(podUrl);
    if (existing && (!existing.expiresAt || existing.expiresAt > new Date())) {
      this.activePodUrl = podUrl;
      return { success: true, session: existing };
    }

    // Try handshake if we have ecosystem identity
    const identity = await this.getEcosystemIdentity();
    if (identity) {
      const handshakeResult = await this.attemptHandshake(podUrl);
      if (handshakeResult.success) {
        return handshakeResult;
      }
    }

    // Fallback to direct pod auth
    return this.directPodAuth(podUrl);
  }

  private async attemptHandshake(podUrl: string): Promise<{ success: boolean; session?: PodSession }> {
    const controlPlaneUrl = process.env.CONTROL_PLANE_URL || 'https://app.synap.live';

    try {
      // Get handshake token from control plane
      const tokenResponse = await fetch(`${controlPlaneUrl}/api/trpc/platform.requestHandshake`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ targetUrl: podUrl }),
        credentials: 'include',
      });

      if (!tokenResponse.ok) {
        log.warn('[AuthManager] Handshake token request failed');
        return { success: false };
      }

      const { result: { data: { token } } } = await tokenResponse.json();

      // Exchange token at pod
      const exchangeResponse = await fetch(`${podUrl}/api/auth/exchange?token=${token}`, {
        credentials: 'include',
      });

      if (!exchangeResponse.ok) {
        log.warn('[AuthManager] Token exchange failed');
        return { success: false };
      }

      // Get session info from pod
      const session = await this.getPodSession(podUrl);
      if (session) {
        this.podSessions.set(podUrl, session);
        this.activePodUrl = podUrl;
        return { success: true, session };
      }

      return { success: false };
    } catch (err) {
      log.error('[AuthManager] Handshake failed:', err);
      return { success: false };
    }
  }

  private async directPodAuth(podUrl: string): Promise<{ success: boolean; session?: PodSession }> {
    // Opens BrowserWindow to pod login (existing openInAppAuth logic)
    return new Promise((resolve) => {
      const authWindow = new BrowserWindow({
        width: 480,
        height: 640,
        webPreferences: {
          session: this.defaultSession,
        }
      });

      authWindow.loadURL(`${podUrl}/.ory/kratos/public/self-service/login/browser`);

      const cookieWatcher = async (_event, cookie, _cause, removed) => {
        if (cookie.name === 'ory_kratos_session' && !removed) {
          this.defaultSession.cookies.off('changed', cookieWatcher);
          authWindow.close();

          const session = await this.getPodSession(podUrl);
          if (session) {
            this.podSessions.set(podUrl, session);
            this.activePodUrl = podUrl;
            resolve({ success: true, session });
          } else {
            resolve({ success: false });
          }
        }
      };

      this.defaultSession.cookies.on('changed', cookieWatcher);

      authWindow.on('closed', () => {
        this.defaultSession.cookies.off('changed', cookieWatcher);
        resolve({ success: false });
      });
    });
  }

  private async getPodSession(podUrl: string): Promise<PodSession | null> {
    try {
      const cookies = await this.defaultSession.cookies.get({
        url: podUrl,
        name: 'ory_kratos_session'
      });

      if (cookies.length === 0) return null;

      // Get session details from pod
      const response = await fetch(`${podUrl}/.ory/kratos/public/sessions/whoami`, {
        headers: { Cookie: `ory_kratos_session=${cookies[0].value}` }
      });

      if (!response.ok) return null;

      const kratosSession = await response.json();

      // Get workspaces from pod
      // This would use the pod's tRPC endpoint
      const workspacesResponse = await fetch(`${podUrl}/api/trpc/workspaces.list`, {
        headers: { Cookie: `ory_kratos_session=${cookies[0].value}` }
      });

      const workspaces = workspacesResponse.ok
        ? (await workspacesResponse.json()).result?.data || []
        : [];

      return {
        podUrl,
        userId: kratosSession.identity.id,
        workspaces,
        sessionCookie: cookies[0].value,
        expiresAt: kratosSession.expires_at ? new Date(kratosSession.expires_at) : undefined,
      };
    } catch (err) {
      log.error('[AuthManager] Failed to get pod session:', err);
      return null;
    }
  }

  // ========== Session Management ==========

  getActivePodSession(): PodSession | null {
    if (!this.activePodUrl) return null;
    return this.podSessions.get(this.activePodUrl) || null;
  }

  getAllPodSessions(): PodSession[] {
    return Array.from(this.podSessions.values());
  }

  async switchToPod(podUrl: string): Promise<boolean> {
    const session = this.podSessions.get(podUrl);
    if (!session) {
      const result = await this.connectToPod(podUrl);
      return result.success;
    }

    this.activePodUrl = podUrl;
    return true;
  }

  async disconnectFromPod(podUrl: string): Promise<void> {
    this.podSessions.delete(podUrl);

    if (this.activePodUrl === podUrl) {
      this.activePodUrl = null;
    }

    // Clear cookies for this pod
    const cookies = await this.defaultSession.cookies.get({ url: podUrl });
    for (const cookie of cookies) {
      await this.defaultSession.cookies.remove(podUrl, cookie.name);
    }
  }

  async signOut(): Promise<void> {
    // Clear all sessions
    this.ecosystemIdentity = null;
    this.podSessions.clear();
    this.activePodUrl = null;

    await this.defaultSession.clearStorageData({ storages: ['cookies'] });
  }
}
```

```typescript
// When switching profiles:
async function switchProfile(profileId: string) {
  const profile = profiles.find(p => p.id === profileId);
  if (!profile) return;

  set({ activeProfileId: profileId });

  // Switch pod connection if needed
  if (profile.podUrl) {
    const authManager = getAuthManager();
```

```typescript
import { useView, useUpdateView, useCreateView } from '@synap/hooks';

export function useHomeView(profile: Profile) {
  // If profile has a homeViewId, load from pod
  const { data: homeView, isLoading } = useView(
    profile.homeViewId || null,
    { enabled: !!profile.homeViewId && !!profile.podUrl }
  );

  const createView = useCreateView();
  const updateView = useUpdateView();

  // Create home view if doesn't exist
  const initializeHomeView = async () => {
    if (profile.homeViewId || !profile.podUrl) return;

    const defaultConfig: BentoViewConfig = {
      blocks: [
        { id: 'welcome', widgetKey: 'welcome', pos: { x: 0, y: 0, w: 12, h: 2 } },
        { id: 'feed', widgetKey: 'feed', pos: { x: 0, y: 2, w: 6, h: 4 } },
        { id: 'quick-access', widgetKey: 'quick-access', pos: { x: 6, y: 2, w: 6, h: 4 } },
      ]
    };

    const view = await createView.mutateAsync({
      name: `Home - ${profile.name}`,
      type: 'bento',
      config: defaultConfig,
      metadata: { isHome: true, profileId: profile.id }
    });

    // Update profile with homeViewId
    useProfileStore.getState().updateProfile(profile.id, {
      homeViewId: view.id
    });

    return view;
  };

  // Save dashboard changes to pod
  const saveDashboard = async (config: BentoViewConfig) => {
    if (!profile.homeViewId) {
      const view = await initializeHomeView();
      if (view) {
        await updateView.mutateAsync({ id: view.id, config });
      }
    } else {
      await updateView.mutateAsync({ id: profile.homeViewId, config });
    }
  };

  return {
    homeView,
    isLoading,
    saveDashboard,
    initializeHomeView,
  };
}
```
